213def strftime(t):
214    """This is the most accurate way to get milliseconds, without microseconds"""

216        return '(null)'

249def make_table(ev):
250    t0 = QTableWidget()
251    t0.horizontalHeader().setStretchLastSection(True)
252    t0.setColumnCount(len(ev.dtype.names))
253    t0.setHorizontalHeaderLabels(ev.dtype.names)
254    t0.verticalHeader().setVisible(False)
255    n_rows = len(ev)
256    t0.setRowCount(n_rows)
257
258    for i0, name in enumerate(ev.dtype.names):
259        for i1 in range(n_rows):
260            v = ev[name][i1]
261
262            if issubdtype(ev.dtype[name].type, floating):
263                v = f'{v:.3f}'
264
265            elif issubdtype(ev.dtype[name].type, integer):
266                v = f'{v}'
267
268            item = QTableWidgetItem(v)
269            item.setFlags(Qt.ItemIsSelectable | Qt.ItemIsEnabled)
270            t0.setItem(i1, i0, item)
271
272    table = t0
273    table.setSizePolicy(QSizePolicy.Minimum, QSizePolicy.Expanding)
274
275    table.setHorizontalScrollBarPolicy(Qt.ScrollBarAlwaysOff)
276
277    table.resizeColumnsToContents()
278    table.setFixedWidth(table.horizontalHeader().length() + table.verticalHeader().width())
279
280    return t0
